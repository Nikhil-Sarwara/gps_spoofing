\section{Introduction}
\begin{frame}{The GPS Spoofing Threat}
\begin{columns}[T]
\column{.55\textwidth}
\begin{itemize}\small
    \item GPS spoofing: counterfeit signals with false coordinates
    \item \textbf{Jamming} = denial; \textbf{Spoofing} = corruption of service
    \item PX4 EKF2 relies heavily on GPS for state estimation
    \item Spoofed signals report healthy fix — attack is silent
    \item Low-cost SDRs $\approx$\$300 make spoofing accessible
\end{itemize}
\vspace{2mm}

\column{.45\textwidth}
\textbf{\small Real-World Incidents}
\begin{itemize}\small
    \item UT Austin 2012 — full 3D UAV control via spoofing
    \item Iran 2019 — claimed capture of US military drone
    \item Commercial drone market projected \$54.6B by 2030
\end{itemize}
\vspace{2mm}
\textbf{\small Why Software-Only?}
\begin{itemize}\small
    \item Hardware solutions (multi-antenna, M-code) too heavy/expensive
    \item Civilian receivers can't access encrypted signals
    \item PX4 already has needed sensors — just need cross-validation
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Research Questions}
\begin{block}{RQ1: GPS Vulnerabilities}
\small What are the primary vulnerabilities of GPS in PX4-based autonomous mobile robots to spoofing attacks?
\end{block}
\begin{block}{RQ2: Sensor Fusion for Real-Time Detection}
\small How can cross-validation between GPS and IMU data effectively detect GPS spoofing in real time, without hardware augmentation?
\end{block}
\begin{block}{RQ3: GPS Trust Index}
\small How can a dynamic, continuous GPS Trust Index be designed to quantify the reliability of GPS data for operational decision-making?

\section{Machine Learning Models}
\begin{frame}{Random Forest \& 1D CNN}
\begin{columns}[T]
\column{.48\textwidth}
\textbf{\small Random Forest}
\begin{itemize}\small
    \item 100 estimators, Gini impurity
    \item \texttt{class\_weight='balanced'}
    \item 1,020-dim flattened input
    \item Feature importance built-in
    \item $<$60s training on M1 Pro, no GPU
    \item Interpretable, robust to class imbalance
\end{itemize}
\column{.48\textwidth}
\textbf{\small 1D CNN}
\begin{itemize}\small
    \item Conv1D(34$\rightarrow$32$\rightarrow$64)
    \item Pool $\rightarrow$ Dense(64$\rightarrow$32)
    \item Dropout(0.3) $\rightarrow$ Sigmoid
    \item 8,000 trainable parameters
    \item Adam optimizer, BCE loss
    \item Early stopping (patience=10)
\end{itemize}
\end{columns}
\vspace{3mm}
\begin{center}
\small\textbf{Dataset: 11,551 windows} (8,085 train / 1,732 val / 1,734 test) $\quad$|\quad 93\% normal / 7\% spoofed
\end{center}
\end{frame}

\section{GPS Trust Index}
\begin{frame}{GPS Trust Index — Graduated Response}
\begin{block}{GTI Mathematical Definition}
\small
\[
G_t = \alpha\, G_{t-1} + (1-\alpha)\bigl[1 - \sigma(k\cdot\Delta\Phi_t - \theta)\bigr]
\]
$\alpha = 0.95$ provides temporal smoothing; $\sigma$ maps physical inconsistency to trust probability $[0,1]$
\end{block}
\begin{center}
\begin{tikzpicture}[node distance=.7cm,
    level/.style={rectangle,draw,text width=8.5cm,text centered,minimum height=.6cm,font=\footnotesize,rounded corners}]
    \node[level,fill=green!8] (g1) {$G \leq 0.7$: Attenuate GPS weight in EKF2};
